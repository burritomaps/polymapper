jQuery examples:

get objects in a bounding box:
 
              $.ajax({ 
                url: "http://civicapi.com/bos_t_stops", 
                dataType: 'jsonp', 
                data: { 
                  "bbox":"-71.062520,42.358028,-71.059516,42.360089" 
                }, 
                success: function(data){ 
                  // manipulate data 
                } 
              });
